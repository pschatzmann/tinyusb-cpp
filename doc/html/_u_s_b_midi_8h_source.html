38 /* A combination of interfaces must have a unique product id, since PC will save device driver after the first plug.
39  * Same VID/PID with different interface e.g MSC (first), then CDC (later) will possibly cause system error on PC.
40  *
41  * Auto ProductID layout's Bitmap:
42  * [MSB] MIDI | HID | MSC | CDC [LSB]
43  */
44 #define _PID_MAP(itf, n) ( (CFG_TUD_##itf) << (n) )
45 #define USB_PID (0x4000 | _PID_MAP(CDC, 0) | _PID_MAP(MSC, 1) | _PID_MAP(HID, 2) | _PID_MAP(MIDI, 3) | _PID_MAP(VENDOR, 4) )
46 #define CONFIG_TOTAL_LEN (TUD_CONFIG_DESC_LEN + TUD_MIDI_DESC_LEN)
47 #if CFG_TUSB_MCU == OPT_MCU_LPC175X_6X || CFG_TUSB_MCU == OPT_MCU_LPC177X_8X || CFG_TUSB_MCU == OPT_MCU_LPC40XX
48  // LPC 17xx and 40xx endpoint type (bulk/interrupt/iso) are fixed by its number
49  // 0 control, 1 In, 2 Bulk, 3 Iso, 4 In etc ...
50  #define EPNUM_MIDI 0x02
51 #else
52  #define EPNUM_MIDI 0x01

194  // Returns the USB string destriptor
195  virtual uint16_t const* get_descriptor_string(uint8_t index, uint16_t langid) {
196 
197  // array of pointer to string descriptors
198  static char const* string_desc_arr [] = {
199  (const char[]) { 0x09, 0x04 }, // 0: is supported language is English (0x0409)
200  "TinyUSB", // 1: Manufacturer
201  "USBMidi", // 2: Product
202  "000001", // 3: Serials, should use chip ID
203  };
204 
205  // on first run we determine the result
206  uint8_t chr_count;
207  if ( index == 0){
208  memcpy(&_desc_str[1], string_desc_arr[0], 2);
209  chr_count = 1;
210  } else {
211  // Note: the 0xEE index string is a Microsoft OS 1.0 Descriptors.
212  // https://docs.microsoft.com/en-us/windows-hardware/drivers/usbcon/microsoft-defined-usb-descriptors
213 
214  if ( !(index < sizeof(string_desc_arr)/sizeof(string_desc_arr[0])) ) return NULL;
215 
216  const char* str = string_desc_arr[index];
217 
218  // Cap at max char
219  chr_count = strlen(str);
220  if ( chr_count > 31 )
221  chr_count = 31;
222 
223  // Convert ASCII string into UTF-16
224  for(uint8_t i=0; i<chr_count; i++){
225  _desc_str[1+i] = str[i];
226  }
227  }
228 
229  // first byte is length (including header), second byte is string type
230  _desc_str[0] = (TUSB_DESC_STRING << 8 ) | (2*chr_count + 2);
231  return _desc_str;
232  }

311  // BLINKING TASK

313  virtual void led_blinking_task(void) {
314  static uint32_t start_ms = 0;
315  static bool led_state = false;
316 
317  // Blink every interval ms
318  if ( board_millis() - start_ms < blinkInteval()) return; // not enough time
319  start_ms += blinkInteval();
320 
321  board_led_write(led_state);
322  led_state = 1 - led_state; // toggle
323  }
